Cerebro vacío — se llenará con la próxima noticia
'; return; } lista.innerHTML = d.registros.slice(0, 15).map(m => { return `
🤖 IA
${m.titulo}
✅
`; }).join(''); } catch(e) { console.error('Error cargando memoria:', e); lista.innerHTML = '
⚠️ Error al leer Supabase
'; } } /* ── REGLA DE SOCIO: AUTO-CARGA AL INICIAR ── */ document.addEventListener('DOMContentLoaded', () => { // Al abrir el panel, traemos la última instrucción de Supabase cargarMemoria(); console.log("🏮 Sala de Redacción conectada a Supabase - mxl"); });
